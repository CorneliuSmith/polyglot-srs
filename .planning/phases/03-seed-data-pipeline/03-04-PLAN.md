## Goal

Create a generic CSV/TSV vocabulary importer that validates incoming data against application schema requirements, provides clear error reporting, and loads validated data into the vocabulary + translations tables. This gives the user a fast path to add/update vocabulary for any language — especially Arabic where curated data will grow over time.

## Success Criteria

1. CLI accepts `python -m backend.services.seeder.run --file vocab.csv --language ar` to import a user CSV
2. Validation catches and reports all schema violations before any DB writes (fail-fast)
3. Validation checks cover: required columns, valid language codes, valid POS values, valid CEFR levels, script validation per language (Arabic words must be Arabic script, Russian must be Cyrillic, English must be Latin), no blank required fields, no duplicate words within the file
4. Clear error report shows line numbers and specific failures so the user can fix their CSV
5. Valid rows are loaded via the same UPSERT path as the language-specific seeders

## Column Spec

The importer accepts CSV or TSV with these columns:

| Column | Required | Description | Validation |
|--------|----------|-------------|------------|
| `word` | Yes | The vocabulary word | Non-empty, correct script for language |
| `definition` | Yes | English definition (or primary definition) | Non-empty |
| `pos` | No | Part of speech | Must be one of: noun, verb, adj, adv, particle, preposition, conjunction, pronoun, interjection |
| `level` | No | CEFR level | Must be A1, A2, B1, B2, C1, or C2 |
| `reading` | No | Pronunciation/transliteration/tashkeel form | Any string |
| `frequency_rank` | No | Frequency ranking | Positive integer |
| `root` | No | Word root (Arabic) | Stored in morphology JSONB |
| `form` | No | Verb form (Arabic I-X) | Stored in morphology JSONB |
| `gender` | No | Grammatical gender | Stored in morphology JSONB |
| `aspect` | No | Verbal aspect (Russian perf/impf) | Stored in morphology JSONB |
| `aspect_partner` | No | Aspect partner word (Russian) | Stored in morphology JSONB |
| `alternatives` | No | Pipe-separated alternative answers | Split on `|`, stored in alternatives array |
| `definition_ru` | No | Russian translation | Stored in translations with locale='ru' |
| `definition_ar` | No | Arabic translation | Stored in translations with locale='ar' |
| `definition_es` | No | Spanish translation | Stored in translations with locale='es' |
| `definition_pt` | No | Portuguese translation | Stored in translations with locale='pt' |

### Task 1: Script validation utilities

Create `backend/services/seeder/validators.py`:

```python
import re
import unicodedata

# Character ranges for script validation
ARABIC_PATTERN = re.compile(r'^[\u0600-\u06FF\u0750-\u077F\u08A0-\u08FF\uFB50-\uFDFF\uFE70-\uFEFF\s\u0640]+$')
CYRILLIC_PATTERN = re.compile(r'^[\u0400-\u04FF\u0500-\u052F\s\-]+$')
LATIN_PATTERN = re.compile(r'^[a-zA-Z\s\-\']+$')

VALID_POS = {"noun", "verb", "adj", "adv", "particle", "preposition", "conjunction", "pronoun", "interjection"}
VALID_LEVELS = {"A1", "A2", "B1", "B2", "C1", "C2"}
SCRIPT_VALIDATORS = {
    "ar": ("Arabic", ARABIC_PATTERN),
    "ru": ("Cyrillic", CYRILLIC_PATTERN),
    "en": ("Latin", LATIN_PATTERN),
}

class ValidationError:
    def __init__(self, row: int, column: str, value: str, message: str):
        self.row = row
        self.column = column
        self.value = value
        self.message = message

    def __str__(self):
        return f"Row {self.row}, column '{self.column}': {self.message} (got: '{self.value}')"

def validate_script(word: str, language_code: str) -> str | None:
    """Returns error message if word contains wrong script, None if OK."""
    if language_code not in SCRIPT_VALIDATORS:
        return None
    script_name, pattern = SCRIPT_VALIDATORS[language_code]
    # Strip diacritics/tashkeel for Arabic before checking
    clean = word.strip()
    if not clean:
        return "word is empty"
    if not pattern.match(clean):
        return f"word contains non-{script_name} characters"
    return None
```

### Task 2: CSV importer implementation

Create `backend/services/seeder/csv_importer.py`:

```python
class CSVImporter(BaseSeeder):
    """Generic CSV/TSV vocabulary importer with validation."""

    def __init__(self, db_url: str, language_code: str, file_path: str):
        super().__init__(db_url)
        self._language_code = language_code
        self.file_path = Path(file_path)
        self.errors: list[ValidationError] = []

    @property
    def language_code(self) -> str:
        return self._language_code

    async def download(self) -> None:
        """No download — file is provided by user."""
        if not self.file_path.exists():
            raise FileNotFoundError(f"Import file not found: {self.file_path}")

    def validate(self, rows: list[dict]) -> tuple[list[dict], list[ValidationError]]:
        """Validate all rows. Returns (valid_rows, errors)."""
        errors = []
        valid = []
        seen_words = set()

        for i, row in enumerate(rows, start=2):  # row 1 is header
            row_errors = []

            # Required fields
            word = row.get("word", "").strip()
            definition = row.get("definition", "").strip()

            if not word:
                row_errors.append(ValidationError(i, "word", "", "required field is empty"))
            if not definition:
                row_errors.append(ValidationError(i, "definition", "", "required field is empty"))

            # Duplicate check within file
            if word and word in seen_words:
                row_errors.append(ValidationError(i, "word", word, "duplicate word in file"))
            seen_words.add(word)

            # Script validation
            if word:
                script_err = validate_script(word, self.language_code)
                if script_err:
                    row_errors.append(ValidationError(i, "word", word, script_err))

            # POS validation
            pos = row.get("pos", "").strip().lower()
            if pos and pos not in VALID_POS:
                row_errors.append(ValidationError(i, "pos", pos, f"invalid POS. Must be one of: {', '.join(sorted(VALID_POS))}"))

            # Level validation
            level = row.get("level", "").strip().upper()
            if level and level not in VALID_LEVELS:
                row_errors.append(ValidationError(i, "level", level, f"invalid CEFR level. Must be one of: {', '.join(sorted(VALID_LEVELS))}"))

            # Frequency rank validation
            freq = row.get("frequency_rank", "").strip()
            if freq:
                if not freq.isdigit() or int(freq) <= 0:
                    row_errors.append(ValidationError(i, "frequency_rank", freq, "must be a positive integer"))

            if row_errors:
                errors.extend(row_errors)
            else:
                valid.append(row)

        return valid, errors

    async def transform(self) -> list[dict]:
        """Parse CSV, validate, return clean records or raise on errors."""
        import csv
        import json

        delimiter = "\t" if self.file_path.suffix == ".tsv" else ","

        with open(self.file_path, encoding="utf-8-sig") as f:
            reader = csv.DictReader(f, delimiter=delimiter)
            raw_rows = list(reader)

        valid_rows, self.errors = self.validate(raw_rows)

        if self.errors:
            # Print error report
            self.logger.error(f"Validation failed with {len(self.errors)} error(s):")
            for err in self.errors:
                self.logger.error(f"  {err}")
            raise ValueError(
                f"CSV validation failed: {len(self.errors)} error(s) in {len(raw_rows)} rows. "
                f"Fix errors and retry. No data was loaded."
            )

        # Transform valid rows to vocabulary records
        records = []
        for row in valid_rows:
            # Build morphology from optional columns
            morphology = {}
            for key in ("root", "form", "gender", "aspect", "aspect_partner"):
                val = row.get(key, "").strip()
                if val:
                    morphology[key] = val

            # Build translations from definition + locale columns
            translations = {"en": row["definition"].strip()}
            for locale in ("ru", "ar", "es", "pt"):
                col = f"definition_{locale}"
                val = row.get(col, "").strip()
                if val:
                    translations[locale] = val

            # Parse alternatives
            alt_str = row.get("alternatives", "").strip()
            alternatives = [a.strip() for a in alt_str.split("|") if a.strip()] if alt_str else []

            freq = row.get("frequency_rank", "").strip()
            freq_rank = int(freq) if freq and freq.isdigit() else None
            level = row.get("level", "").strip().upper() or self.rank_to_level(freq_rank)

            records.append({
                "word": row["word"].strip(),
                "reading": row.get("reading", "").strip() or None,
                "pos": row.get("pos", "").strip().lower() or None,
                "level": level or None,
                "frequency_rank": freq_rank,
                "morphology": json.dumps(morphology, ensure_ascii=False) if morphology else "{}",
                "translations": translations,
            })

        self.logger.info(f"Validated and transformed {len(records)} records from {self.file_path.name}")
        return records
```
