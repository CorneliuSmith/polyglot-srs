### Task 2: English seeder implementation

Create `backend/services/seeder/seed_english.py`:

```python
class EnglishSeeder(BaseSeeder):
    language_code = "en"

    async def download(self) -> None:
        """Download NLTK WordNet corpus if not present."""
        import nltk
        try:
            nltk.data.find("corpora/wordnet")
        except LookupError:
            self.logger.info("Downloading WordNet corpus...")
            nltk.download("wordnet", quiet=True)

    async def transform(self) -> list[dict]:
        """Merge frequency list with WordNet definitions."""
        import csv
        import json
        from nltk.corpus import wordnet as wn

        # Read frequency list
        freq_path = DATA_DIR / "en_frequency.tsv"
        if not freq_path.exists():
            raise FileNotFoundError(f"English frequency file not found at {freq_path}")

        freq_words = []
        with open(freq_path, encoding="utf-8") as f:
            reader = csv.DictReader(f, delimiter="\t")
            for row in reader:
                freq_words.append((int(row["rank"]), row["word"].strip()))

        # Optional: spaCy for POS
        nlp = None
        try:
            import spacy
            nlp = spacy.load("en_core_web_sm")
        except (ImportError, OSError):
            self.logger.warning("spaCy not available — POS from WordNet only")

        records = []
        for rank, word in freq_words:
            # Skip function words (the, a, is, etc.) for vocabulary study
            synsets = wn.synsets(word)
            if not synsets:
                continue

            # Use first synset for primary definition
            primary = synsets[0]
            definition = primary.definition()
            pos_map = {"n": "noun", "v": "verb", "a": "adj", "r": "adv", "s": "adj"}
            wn_pos = pos_map.get(primary.pos(), None)

            # spaCy POS if available
            pos = wn_pos
            morphology = {"lemma": word}
            if nlp:
                try:
                    doc = nlp(word)
                    if doc:
                        token = doc[0]
                        pos = token.pos_.lower() if token.pos_ else wn_pos
                        morphology["lemma"] = token.lemma_
                except Exception:
                    pass

            records.append({
                "word": word,
                "reading": None,
                "pos": pos,
                "level": self.rank_to_level(rank),
                "frequency_rank": rank,
                "morphology": json.dumps(morphology),
                "translations": {"en": definition},
            })

        self.logger.info(f"Transformed {len(records)} English words")
        return records
```

- Test that `language_code` property is correct for each
- Test that `transform()` output from each seeder matches expected schema:

  - Every record has `translations` dict with at least one entry
  - `morphology` is valid JSON string
  - `level` is None or valid CEFR level
  - `frequency_rank` is None or positive int
- Test CLI runner module can be imported without errors
- Validate no duplicate words in transform output
